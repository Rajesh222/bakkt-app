Overview

Bakkt uses webhooks to provide real-time, asynchronous updates on KYC/KYB verification status and other onboarding events.

Setup

Configure your webhook endpoint using the PATCH /merchant endpoint:

  • Webhook URL: Your server endpoint where Bakkt will send notifications
  • Webhook Secret: Shared secret for verifying webhook authenticity
  • Environment: Configure separately for Sandbox and Production
curl -X PATCH https://api.bakkt.com/merchant \
  -H "Authorization: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "webhook": "https://your-domain.com/webhooks",
    "webhook_secret": "your_secure_secret"
  }'
const response = await fetch('https://api.bakkt.com/merchant', {
  method: 'PATCH',
  headers: {
    'Authorization': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    webhook: 'https://your-domain.com/webhooks',
    webhook_secret: 'your_secure_secret'
  })
});
❗️

Production vs Sandbox: Configure webhook URL separately for each environment. Use different secrets for each environment for security.

Verifying Webhook Authenticity

Every webhook includes an Authorization header with your webhook secret. Always verify this before processing:

function verifyWebhook(req, expectedSecret) {
  const authHeader = req.headers['authorization'];
  const receivedSecret = authHeader?.replace('API-Key ', '');

  if (receivedSecret !== expectedSecret) {
    throw new Error('Invalid webhook signature');
  }
}

Complete Express.js Example:

const express = require('express');
const app = express();

app.post('/webhooks', express.json(), (req, res) => {
  try {
    // 1. Verify authenticity
    verifyWebhook(req, process.env.BAKKT_WEBHOOK_SECRET);

    // 2. Process webhook
    const { type, subType, uuid, data } = req.body;

    console.log(`Received ${type} webhook with subType: ${subType}`);
    console.log('User/Corporate UUID:', uuid);

    // Handle different webhook types
    switch(type) {
      case 'KYC':
        handleKYCWebhook(subType, uuid, data);
        break;
      case 'KYB':
        handleKYBWebhook(subType, uuid, data);
        break;
      default:
        console.log('Unknown webhook type:', type);
    }

    // 3. Respond with 200 to acknowledge receipt
    res.status(200).send('OK');
  } catch (error) {
    console.error('Webhook error:', error);
    res.status(400).send('Bad Request');
  }
});

Retry Logic

Bakkt will retry failed webhook deliveries up to 7 times with exponential backoff:

  • After 1st failure: 3 seconds
  • After 2nd failure: ~1 minute
  • After 3rd failure: ~12 minutes
  • After 4th failure: ~1 hour
  • After 5th failure: ~4 hours
  • After 6th failure: ~12 hours
  • After 7th failure: ~33 hours

Ensure your endpoint responds with 2xx status codes to avoid unnecessary retries.

❗️

Important: If a webhook for a status update is not received and a newer status is reached later, the older status will not be sent again. Webhooks always reflect the latest state only

📘

Idempotency: Design your webhook handler to be idempotent. Bakkt may send the same webhook multiple times if there are delivery issues.

KYC Webhook Handler

function handleKYCWebhook(subType, userUuid, data) {
  switch(subType) {
    case 'FULL_USER':
      // KYC approved
      console.log(`User ${userUuid} KYC approved`);
      // Update user status in your database
      db.users.update(userUuid, { kyc_status: 'approved', can_transact: true });
      // Send notification to user
      sendEmail(userUuid, 'KYC Approved', 'You can now start transacting!');
      break;

    case 'SOFT_KYC_FAILED':
      // User can retry
      console.log(`User ${userUuid} KYC failed (can retry)`);
      console.log('Rejection reason:', data.rejectionReason);
      // Notify user with guidance
      sendEmail(userUuid, 'KYC Needs Correction', data.rejectionMessage);
      break;

    case 'HARD_KYC_FAILED':
      // Permanent failure
      console.log(`User ${userUuid} KYC permanently failed`);
      // Contact support flow
      notifySupport(userUuid);
      break;

    case 'KYC_PENDING':
      // Under review
      console.log(`User ${userUuid} KYC under review`);
      // Update UI to show "Verification in progress"
      break;

    default:
      console.log(`Unknown KYC subType: ${subType}`);
  }
}

Complete Webhook Handler

Full implementation with all webhook types:

const express = require('express');
const crypto = require('crypto');
const app = express();

// Middleware to verify webhook signature
function verifyWebhookSignature(req, res, next) {
  const receivedSecret = req.headers['authorization']?.replace('API-Key ', '');
  const expectedSecret = process.env.BAKKT_WEBHOOK_SECRET;

  if (receivedSecret !== expectedSecret) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  next();
}

// Webhook endpoint
app.post('/webhooks/bakkt',
  express.json(),
  verifyWebhookSignature,
  async (req, res) => {
    const { type, subType, uuid, data } = req.body;

    console.log(`[Webhook] ${type} - ${subType} for ${uuid}`);

    try {
      // Route to appropriate handler
      switch(type) {
        case 'KYC':
          await handleKYC(subType, uuid, data);
          break;
        case 'KYB':
          await handleKYB(subType, uuid, data);
          break;
        case 'fiatToCrypto':
        case 'cryptoToFiat':
          // Transaction webhooks (see Stablecoin API guide)
          await handleTransaction(type, subType, uuid, data);
          break;
        default:
          console.log('Unknown webhook type:', type);
      }

      // Acknowledge receipt
      res.status(200).send('OK');
    } catch (error) {
      console.error('Error processing webhook:', error);
      // Still return 200 to prevent retries for processing errors
      res.status(200).send('OK');
    }
  }
);

async function handleKYC(subType, userUuid, data) {
  // Update user in database
  await db.users.update(userUuid, {
    kyc_status: subType,
    kyc_updated_at: new Date()
  });

  // Notify user based on status
  if (subType === 'FULL_USER') {
    await sendUserNotification(userUuid, {
      title: 'Verification Complete',
      body: 'Your account is now fully verified!'
    });
  } else if (subType.includes('FAILED')) {
    await sendUserNotification(userUuid, {
      title: 'Verification Failed',
      body: data.rejectionMessage || 'Please contact support'
    });
  }
}

async function handleKYB(subType, corporateUuid, data) {
  // Update corporate in database
  await db.corporates.update(corporateUuid, {
    kyb_status: subType,
    kyb_updated_at: new Date()
  });

  // Notify corporate admin
  if (subType === 'ACTIVE') {
    await sendCorporateNotification(corporateUuid, {
      title: 'Business Verification Complete',
      body: 'Your company can now start using Bakkt services!'
    });
  }
}

app.listen(3000, () => console.log('Webhook server running on port 3000'));

Webhook Security Best Practices

Signature Verification

Always verify webhook signatures

// GOOD
function handleWebhook(req, res) {
  if (!verifySignature(req)) {
    return res.status(401).send('Unauthorized');
  }
  // Process webhook...
}

// BAD - Never do this
function handleWebhook(req, res) {
  // Process without verification - INSECURE!
}

Idempotency

Handle duplicate webhooks gracefully

const processedWebhooks = new Set();

function handleWebhook(webhookId, data) {
  // Check if already processed
  if (processedWebhooks.has(webhookId)) {
    console.log('Duplicate webhook, skipping');
    return;
  }

  // Process webhook
  processWebhookData(data);

  // Mark as processed
  processedWebhooks.add(webhookId);
}

Use webhook ID or combination of (type + subType + uuid + timestamp) as idempotency key.

Error Handling

Respond with 2xx even if processing fails

app.post('/webhooks', async (req, res) => {
  try {
    // Verify signature first
    verifyWebhook(req, WEBHOOK_SECRET);

    // Acknowledge immediately
    res.status(200).send('OK');

    // Process asynchronously (don't block response)
    processWebhookAsync(req.body).catch(error => {
      console.error('Async processing error:', error);
      // Handle error internally, don't fail webhook
    });
  } catch (error) {
    // Only reject if signature invalid
    if (error.message.includes('signature')) {
      res.status(401).send('Unauthorized');
    } else {
      // Accept webhook but log error
      console.error('Webhook error:', error);
      res.status(200).send('OK');
    }
  }
});

Rate Limiting

Protect your webhook endpoint

const rateLimit = require('express-rate-limit');

const webhookLimiter = rateLimit({
  windowMs: 1 * 60 * 1000, // 1 minute
  max: 100, // 100 requests per minute
  message: 'Too many webhook requests'
});

app.post('/webhooks', webhookLimiter, handleWebhook);

Log all webhook events

Webhook Debugging Checklist

When webhooks aren't being received:

  • Is webhook URL configured? (Check with GET /merchant)
  • Is webhook URL publicly accessible? (Test with curl)
  • Is endpoint returning 2xx status codes? (Check logs)
  • Is signature verification passing? (Log auth header)
  • Is firewall/WAF blocking requests? (Check network logs)
  • Is SSL certificate valid? (Bakkt requires valid HTTPS)
  • Are you checking the right environment? (Sandbox vs Production)
